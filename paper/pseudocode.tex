    \underline{function Euclid} $(a,b)$\;
    \Input{Probability distribution over target categories of all images for one product}
    \Output{Category prediction of the product}
    \eIf{$b=0$}
      {
        return $a$\;
      }
      {
        return Euclid$(b,a\mod b)$\;
      }
